const SHEET_ID         = '1hVbwRJAU2d0ozTPGfDrePpgPxG8kjWWVrI7_9eHJTZg';
const TARGETS_SHEET_ID = '1MLEeMUgTnKgNkkpbkLv4BI6Gzwwk7F4FL6zYX3sN4_c';
const LOB_NESTING_TABS = ['R1 Nesting Raw','Supply Nesting Raw','R2 Nesting Raw','Exec. Escalations Nesting Raw','AirCover Nesting Raw','Safety Essentials Nesting Raw','Safety Advanced Nesting Raw'];

function doGet(e) {
  let payload;
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    payload = {
      lastSync:   new Date().toISOString(),
      batches:    parseBatches(ss),
      trainees:   parseTrainees(ss),
      nesting:    parseNesting(ss),
      nestingRaw: parseAllNestingRaw(ss),
      targets:    parseTargets()
    };
  } catch(err) {
    payload = { error: err.message };
  }
  const json = JSON.stringify(payload);
  // JSONP support — when ?callback=fn is passed, return JavaScript that calls fn(data)
  // This bypasses CORS for Workspace deployments that don't send Access-Control headers.
  const cb = e && e.parameter && e.parameter.callback;
  if (cb && /^[\w$.]+$/.test(cb)) {
    return ContentService.createTextOutput(cb + '(' + json + ');')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function parseBatches(ss) {
  const sh = ss.getSheetByName('Batch Throughput 2026');
  if (!sh) return [];
  const rows = sh.getDataRange().getValues();
  const out = [];
  for (let i = 1; i < rows.length; i++) {
    const r = rows[i];
    const code = String(r[5]||'').trim();
    if (!code || code === 'Batch Code') continue;
    const pct = v => { const s=String(v||'').replace('%','').trim(); const n=parseFloat(s); return isNaN(n)?null:n; };
    const num = v => { const n=parseFloat(String(v||'')); return isNaN(n)?null:n; };
    const dt  = v => { if(!v) return null; if(v instanceof Date) return Utilities.formatDate(v,'UTC','yyyy-MM-dd'); return String(v).trim()||null; };
    out.push({
      waveCode: code, lob: String(r[1]||'').trim(),
      batchType: String(r[2]||'').trim(), trainer: String(r[4]||'').trim(),
      startDate: dt(r[6]), trainingEndDate: dt(r[7]),
      alStartDate: dt(r[8]), alEndDate: dt(r[9]),
      initialHC: num(r[10]), transferOutHC: num(r[11]), transferInHC: num(r[12]),
      finalTrainingHC: num(r[13]), supportGroupHC: num(r[14]),
      resignedHC: num(r[15]), terminatedHC: num(r[16]),
      firstPassHC: num(r[17]), handoverToNestingHC: num(r[19]),
      trainingThroughput: pct(r[20]), trainingStatus: String(r[21]||'').trim(),
      nestingStartDate: dt(r[22]), nestingEndDate: dt(r[23]),
      finalNestingHC: num(r[28]),
      nestingThroughput: pct(r[35]), overallThroughput: pct(r[36]),
      finalNestingStatus: String(r[37]||'').trim(), handoverToOps: num(r[34]),
      notes: String(r[39]||'').trim()
    });
  }
  return out;
}

function parseTrainees(ss) {
  const sh = ss.getSheetByName('Batch Info 2026');
  if (!sh) return [];
  const rows = sh.getDataRange().getValues();
  const out = [];
  for (let i = 1; i < rows.length; i++) {
    const r = rows[i];
    const name = String(r[0]||'').trim();
    if (!name || name === 'Name of Trainee') continue;
    const pct = v => { const s=String(v||'').replace('%','').trim(); const n=parseFloat(s); return isNaN(n)?null:n; };
    const dt  = v => { if(!v) return null; if(v instanceof Date) return Utilities.formatDate(v,'UTC','yyyy-MM-dd'); return String(v).trim()||null; };
    out.push({
      name, email: String(r[1]||'').trim(), role: String(r[2]||'').trim(),
      lob: String(r[3]||'').trim(), orgJoiningDate: dt(r[4]),
      employeeType: String(r[5]||'').trim(),
      batchCode: String(r[8]||'').trim(), trainer: String(r[9]||'').trim(),
      trainingStatus: String(r[10]||'').trim(), comments: String(r[11]||'').trim(),
      score1: pct(r[12]), score2: pct(r[13]), passing: pct(r[14]), score: pct(r[15]),
      nestingStartDate: dt(r[16]), nestingEndDate: dt(r[17]),
      nestingStatus: String(r[18]||'').trim(), finalNestingStatus: String(r[21]||'').trim()

function parseNesting(ss) {
  const sh = ss.getSheetByName('Nesting Dashboard');
  if (!sh) return [];
  const rows = sh.getDataRange().getValues();
  const out = [];
  const pct = v => { const s=String(v||'').replace('%','').trim(); const n=parseFloat(s); return isNaN(n)?null:n; };
  const num = v => { const n=parseFloat(String(v||'')); return isNaN(n)?null:n; };
  const dt  = v => { if(!v) return null; if(v instanceof Date) return Utilities.formatDate(v,'UTC','yyyy-MM-dd'); return String(v).trim()||null; };
  for (let i = 1; i < rows.length; i++) {
    const r = rows[i];
    const name = String(r[6]||'').trim();
    if (!name || name === 'Ambassador name') continue;
    out.push({
      nestingStartDate: dt(r[1]),
      nestingEndDate:   dt(r[2]),
      lob:              String(r[3]||'').trim(),
      batchCode:        String(r[4]||'').trim(),
      trainer:          String(r[5]||'').trim(),
      name:             name,
      nestingGAPct:     pct(r[7]),
      nestingMetricsMet:num(r[8]),
      nestingStatus:    String(r[9]||'').trim()
    });
  }
  return out;
}

// ── Helpers shared by raw + targets parsers ──
function _gsNum(v) {
  if (v === '' || v == null) return null;
  if (typeof v === 'number') return v;
  let s = String(v).replace('%','').replace('$','').trim();
  if (!s || s.toUpperCase() === 'TBD') return null;
  const n = parseFloat(s);
  return isNaN(n) ? null : n;
}
function _gsPct(v) {
  if (v === '' || v == null) return null;
  if (typeof v === 'number') return v < 1 ? v * 100 : v;
  const s = String(v).replace('%','').trim();
  if (!s) return null;
  const n = parseFloat(s); return isNaN(n) ? null : n;
}
function _gsDate(v) {
  if (!v) return null;
  if (v instanceof Date) return Utilities.formatDate(v,'UTC','yyyy-MM-dd');
  return String(v).trim() || null;
}
function _cleanMetric(s) { return String(s||'').replace(/\s*\([^)]*\)\s*$/,'').trim(); }

// Parses ALL LOB nesting raw tabs (different layouts handled dynamically)
function parseAllNestingRaw(ss) {
  const out = {};
  for (const tab of LOB_NESTING_TABS) {
    const sh = ss.getSheetByName(tab);
    out[tab] = sh ? parseLobNestingTab(sh) : [];
  }
  return out;
}

// Parses one Nesting Raw tab — detects column structure from headers
function parseLobNestingTab(sh) {
  const rows = sh.getDataRange().getValues();
  if (rows.length < 3) return [];
  const secRow = rows[0];
  const hdrRow = rows[1];

  // Identify trainee info columns by header name
  const col = {};
  for (let c = 0; c < hdrRow.length; c++) {
    const h = String(hdrRow[c]||'').trim().toLowerCase();
    if (!h) continue;
    if (h.includes('name of trainee'))   col.name = c;
    else if (h === 'lob')                col.lob = c;
    else if (h === 'nesting start date') col.startDate = c;
    else if (h === 'nesting end date')   col.endDate = c;
    else if (h.includes('batch code'))   col.batchCode = c;
    else if (h === 'trainer')            col.trainer = c;
    else if (h === 'nesting status')     col.status = c;
  }

  // Detect sections from row 1 (merged cells leave blanks; section starts at non-empty cell)
  const sections = []; let cur = null;
  for (let c = 0; c < secRow.length; c++) {
    const s = String(secRow[c]||'').trim();
    if (s) { if (cur) { cur.end = c-1; sections.push(cur); } cur = { name:s, start:c, end:-1, metrics:[] }; }
  }
  if (cur) { cur.end = secRow.length-1; sections.push(cur); }

  // Detect weekly GA columns and per-week metrics
  const weeklyGACols = {};
  sections.forEach(sec => {
    for (let c = sec.start; c <= sec.end; c++) {
      const h = String(hdrRow[c]||'').trim();
      if (!h) continue;
      if (Object.values(col).indexOf(c) >= 0) continue;
      const isGASection = /w-?o-?w ga|^\d+-\d+$|overall/i.test(sec.name);
      if (isGASection) weeklyGACols[h.trim()] = c;
      else sec.metrics.push({ name: _cleanMetric(h), col: c });
    }
  });

  // Read trainee rows
  const out = [];
  for (let r = 2; r < rows.length; r++) {
    const row = rows[r];
    const name = col.name != null ? String(row[col.name]||'').trim() : '';
    if (!name) continue;
    const t = {
      name,
      lob:              col.lob       != null ? String(row[col.lob]||'').trim() : '',
      nestingStartDate: col.startDate != null ? _gsDate(row[col.startDate]) : null,
      nestingEndDate:   col.endDate   != null ? _gsDate(row[col.endDate])   : null,
      batchCode:        col.batchCode != null ? String(row[col.batchCode]||'').trim() : '',
      trainer:          col.trainer   != null ? String(row[col.trainer]||'').trim() : '',
      nestingStatus:    col.status    != null ? String(row[col.status]||'').trim() : '',
      weeklyGA: {},
      weeks: {}
    };
    Object.keys(weeklyGACols).forEach(k => { const v = _gsPct(row[weeklyGACols[k]]); if (v != null) t.weeklyGA[k] = v; });
    sections.forEach(sec => {
      if (!sec.metrics.length) return;
      const data = {};
      sec.metrics.forEach(m => {
        const raw = row[m.col];
        if (raw !== '' && raw != null) data[m.name] = _gsNum(raw);
      });
      if (Object.keys(data).length) t.weeks[sec.name] = data;
    });
    out.push(t);
  }
  return out;
}

// Parses the targets sheet (block-style: LOB header → 'Metrics|Weightage|Week 1..|Cumulative' → metric rows → blank → next LOB)
function parseTargets() {
  let ss; try { ss = SpreadsheetApp.openById(TARGETS_SHEET_ID); } catch(e) { return { error:'Targets sheet not accessible: '+e.message }; }
  const sheets = ss.getSheets();
  const sh = sheets.find(s => s.getSheetId() === 1862855243) || sheets[0];
  const rows = sh.getDataRange().getValues();
  const lobs = {};
  let curLob = null, weekCols = [], cumCol = -1, wtCol = -1, mode = 'idle';
  for (let r = 0; r < rows.length; r++) {
    const row = rows[r];
    const c0 = String(row[0]||'').trim();
    const c1 = String(row[1]||'').trim();
    const nonEmpty = row.filter(c => c !== '' && c != null).length;

    // Detect LOB header (single populated cell)
    if (c0 && nonEmpty === 1 && !/^metrics?$/i.test(c0)) {
      curLob = c0;
      if (!lobs[curLob]) lobs[curLob] = { weights:{}, weeks:{} };
      mode = 'expectHeader';
      continue;
    }
    if (mode === 'expectHeader' && /^metrics?$/i.test(c0) && /weightage/i.test(c1)) {
      weekCols = []; cumCol = -1; wtCol = 1;
      for (let c = 2; c < row.length; c++) {
        const h = String(row[c]||'').trim().toLowerCase();
        if (/week\s*[-\s]*\d+/.test(h)) weekCols.push(c);
        else if (h.includes('cumulative')) cumCol = c;
      }
      mode = 'parsing';
      continue;
    }
    if (mode === 'parsing' && curLob) {
      if (!c0) { mode = 'idle'; continue; }
      const metric = c0;
      const wt = _gsPct(row[wtCol]);
      if (wt != null) lobs[curLob].weights[metric] = wt;
      weekCols.forEach((c, idx) => {
        const wk = idx + 1;
        if (!lobs[curLob].weeks[wk]) lobs[curLob].weeks[wk] = {};
        const v = _gsNum(row[c]);
        if (v != null) lobs[curLob].weeks[wk][metric] = v;
      });
      if (cumCol >= 0) {
        if (!lobs[curLob].weeks.cumulative) lobs[curLob].weeks.cumulative = {};
        const v = _gsNum(row[cumCol]);
        if (v != null) lobs[curLob].weeks.cumulative[metric] = v;
      }
    }
  }
  return lobs;
}
